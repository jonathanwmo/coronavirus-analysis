16   def find_dates(graph_type):
17       ''' 
18       finds the dates in ourworldindata.org csv file 
19       :param graph_type: string of graph type 
20       :return: a list of dates 
21       '''
22       datafile = urllib.request.urlopen('https://covid.ourworldindata.org/data/ecdc/' + graph_type)
23       dates_list = []
24       for line in datafile.readlines():
25           line = line.decode('utf-8').strip()
26           row = line.split(",")
27   
28           dates_list.append(row[0])
29       datafile.close()
30       return dates_list

32   def find_cases(column, graph_type):
33       ''' 
34       finds total cases for given population (world or country) 
35       :param column: the column number for csv file 
36       :param graph_type: string of graph type 
37       :return: a list of total cases for inputted location 
38       '''
39       datafile = urllib.request.urlopen('https://covid.ourworldindata.org/data/ecdc/' + graph_type)
40       num_cases = []
41       for line in datafile.readlines():
42           line = line.decode('utf-8').strip()
43           row = line.split(",")
44   
45           num_cases.append(row[column])
46       datafile.close()
47       return num_cases

49   def find_index(country, graph_type):
50       ''' 
51       finds index of country in csv file 
52       :param country: string of inputted country 
53       :param graph_type: string of graph type 
54       :return: index (column) that the country is located in the csv file 
55       '''
56       datafile = urllib.request.urlopen('https://covid.ourworldindata.org/data/ecdc/' + graph_type)
57       line = datafile.readline()
58       line = line.decode('utf-8').strip()
59       countries_list = line.split(",")
60       index = countries_list.index(country)
61       return index
