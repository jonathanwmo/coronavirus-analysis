38   def find_cases(column: int, graph_type: str):
39       ''' 
40       finds total cases for given population (world or country) 
41       :param column: the column number for csv file 
42       :param graph_type: string of graph type 
43       :return: a list of total cases (or deaths) for inputted location 
44       '''
45   
46       num_cases = []
47       with open(os.path.join(BASE_DIR, 'csvs/{}'.format(graph_type)), 'r') as f:
48           for line in f.readlines():
49               row = line.strip().split(",")
50               num_cases.append(row[column])
51           f.close()
52       return num_cases

55   def find_index(country: str, graph_type: str):
56       ''' 
57       finds index of country in csv file 
58       :param country: string of inputted country 
59       :param graph_type: string of graph type 
60       :return: index (column) that the country is located in the csv file 
61       '''
62       with open(os.path.join(BASE_DIR, 'csvs/{}'.format(graph_type)), 'r') as f:
63           line = f.readline()
64           countries_list = line.strip().split(",")
65           index = countries_list.index(country)
66           f.close()
67       return index

72       '''

94           country = country.replace("Of", "of")
95       if " The" in country:
96           country = country.replace("The", "the")

98           country = "Sint Maarten (Dutch part)"

165      elif highest_cases <= 500000:

167      elif highest_cases <= 1000000:

169      elif highest_cases <= 2500000:

235          transparent=True)

238      my_graph.clf()
239      my_graph.close('all')
