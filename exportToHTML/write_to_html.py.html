15   def get_country_data(country: str):
16       ''' 
17       looks at worldometers.info to get values of Total Cases, New Cases, Total Deaths, New Deaths, 
18       Total Recovered, Active Cases, Total Cases per 1 Million People, Deaths per 1 Million People, 
19       Total Tests, Tests per 1 Million People, Population 
20       Use Beautiful Soup as well as regular expressions to parse html pages 
21       :param country: string of inputted country 
22       :return: a list of 10 strings which represent integer values of said values mentioned earlier 
23       '''
24       # handle edge cases and countries that contain more than 1 word, when
25       # parsing worldometers.info spaces are replaced by '-'
26       country = country.lower().replace(" ", "-")
27       if country == "united-states":
28           country = "us"
29       elif country == "united-kingdom":
30           country = "uk"
31   
32       req = Request('https://www.worldometers.info/coronavirus/',
33                     headers={'User-Agent': 'Mozilla/5.0'})
34       page = urlopen(req).read()
35       page = BeautifulSoup(page, 'html.parser')
36   
37       # create a list of each line of the html page
38       page_list = str(page).split("\n")
39   
40       # country_numbers_html list will contain the desired ~10 lines of html
41       # text for given country
42       country_numbers_html = []
43       # country_numbers list will contain solely the 10 desired values extracted
44       # from country_numbers_html using regular expressions
45       country_numbers = []
46   
47       inlines = False
48       if country == "world":  # special case for world data as its formatted differently
49           for line in page_list:
50               if ('">World') in str(line):
51                   inlines = True
52               # when inlines=True, we are in desired lines of html text for world
53               if inlines:
54                   country_numbers_html.append(line)
55               # 'data-continent...' is our list line of desired html text
56               if ('data-continent="all"') in str(line):
57                   break
58           # don't need number in first extracted line of html text
59           country_numbers_html = country_numbers_html[1:]
60   
61           for i in country_numbers_html:
62               finder = re.compile(r'\b\d[\d,.]*\b')
63               if re.search(finder, i) is not None:
64                   country_numbers.append(re.search(finder, i).group())
65               else:
66                   # if no match, append an empty string to country_numbers list instead
67                   country_numbers.append("")
68           # won't display 5th and 6th value in my table
69           country_numbers.pop(5)
70           country_numbers.pop(5)
71           country_numbers[-2] = '7,785,879,495'
72   
73       else:
74           for line in page_list:
75               if ("country/" + country + "/") in str(line.strip()):
76                   # when inlines=True, we are in desired lines of html text for
77                   # given country
78                   inlines = True
79               if inlines:
80                   country_numbers_html.append(line)
81               # 'a href....' is our list line of desired html text
82               if ('a href="/world-population/' +
83                       country + "-population/") in str(line):
84                   break
85           # don't need number in first extracted line of html text
86           country_numbers_html = country_numbers_html[1:]
87   
88           for i in country_numbers_html:
89               finder = re.compile(r'\b\d[\d,.]*\b')
90               if re.search(finder, i) is not None:
91                   country_numbers.append(re.search(finder, i).group())
92               else:
93                   # if no match, append an empty string to country_numbers list
94                   # instead
95                   country_numbers.append("")
96           # 3rd index is incorrect line spacing on html page so must be taken out
97           # of list
98           country_numbers.pop(3)
99           # won't display 5th and 6th value in my table
100          country_numbers.pop(5)
101          country_numbers.pop(5)
102      country_numbers = country_numbers[0:10]
103      return country_numbers

106  def update_country_html(country: str):
107      ''' 
108      for a given country passed in, this function writes to that country's html page 
109      to update the data values and which .png image to pull from a given date 
110      :param country: string of country 
111      :return: a list of 10 strings which represent integer values 
112      '''
113  
114      # handle edge cases when countries are named differently across different websites
115      if country == 'Bonaire Sint Eustatius and Saba':
116          country_numbers = get_country_data('caribbean-netherlands')
117      elif country == 'Brunei':
118          country_numbers = get_country_data('brunei-darussalam')
119      elif country == 'Cape Verde':
120          country_numbers = get_country_data('cabo-verde')
121      elif country == "Cote d'Ivoire":
122          country_numbers = get_country_data('cote-d-ivoire')
123      elif country == 'Democratic Republic of Congo':
124          country_numbers = get_country_data('democratic-republic-of-the-congo')
125      elif country == 'Falkland Islands':
126          country_numbers = get_country_data('falkland-islands-malvinas')
127      elif country == 'Guernsey':
128          country_numbers = get_country_data('channel-islands')
129      elif country == 'Palestine':
130          country_numbers = get_country_data('state-of-palestine')
131      elif country == 'Sint Maarten (Dutch part)':
132          country_numbers = get_country_data('sint-maarten')
133      elif country == 'Timor':
134          country_numbers = get_country_data('timor-leste')
135      elif country == 'Vatican':
136          country_numbers = get_country_data('holy-see')
137      elif country == 'Vietnam':
138          country_numbers = get_country_data('viet-nam')
139      else:
140          country_numbers = get_country_data(country)
141  
142      _country_ = country.lower().replace(" ", "_")
143      country = country.lower().replace(" ", "")
144  
145      dir_path = os.path.dirname(os.path.realpath(__file__))[
146          0:56] + "/src/htmls/" + country + ".html"
147  
148      # create the dates of the past 3 days as strings
149      date_aot = str(datetime.today())[0:10]
150      date_aoy = str(datetime.today() - timedelta(days=1))[0:10]
151      date_ao2 = str(str(datetime.today() - timedelta(days=2))[0:10])
152      date_ao3 = str(str(datetime.today() - timedelta(days=2))[0:10])
153  
154      # replace any empty values with n/a
155      for i in range(len(country_numbers)):
156          if country_numbers[i] == "":
157              country_numbers[i] = "n/a"
158  
159      # open country html file for reading
160      with open(dir_path, 'r') as f:
161          newstr = """"""
162          for line in f.readlines():
163              # update the dates
164              if date_aoy in line:
165                  newstr += line.replace(date_aoy, date_aot)
166              elif date_ao2 in line:
167                  newstr += line.replace(date_ao2, date_aot)
168              elif date_ao3 in line:
169                  newstr += line.replace(date_ao3, date_aot)
170              # update the values on html page as scraped from worldometer.info
171              # site
172              elif '<p><b><u>Total Cases</u></b><br><br> ' in line:
173                  newstr += '                    <p><b><u>Total Cases</u></b><br><br> ' + \
174                      country_numbers[0] + '</p>\n'
175              elif '<p><b><u>New Cases </u></b><br><br>' in line:
176                  newstr += '                    <p><b><u>New Cases </u></b><br><br>' + \
177                      country_numbers[1] + '</p>\n'
178              elif '<p><b><u>Cases/1 Million </u></b><b' in line:
179                  newstr += '                    <p><b><u>Cases/1 Million </u></b><br><br>' + \
180                      country_numbers[5] + '</p>\n'
181              elif '<p><b><u>Total Deaths </u></b><br><' in line:
182                  newstr += '                    <p><b><u>Total Deaths </u></b><br><br>' + \
183                      country_numbers[2] + '</p>\n'
184              elif '<p><b><u>New Deaths </u></b><br><br' in line:
185                  newstr += '                    <p><b><u>New Deaths </u></b><br><br>' + \
186                      country_numbers[3] + '</p>\n'
187              elif '<p><b><u>Deaths/1 Million</u></b><b' in line:
188                  newstr += '                    <p><b><u>Deaths/1 Million</u></b><br><br>' + \
189                      country_numbers[6] + '</p>\n'
190              elif '<p><b><u>Total Recovered </u></b><b' in line:
191                  newstr += '                    <p><b><u>Total Recovered </u></b><br><br>' + \
192                      country_numbers[4] + '</p>\n'
193              elif '<p><b><u>Total Tests </u></b><br><b' in line:
194                  newstr += '                    <p><b><u>Total Tests </u></b><br><br>' + \
195                      country_numbers[7] + '</p>\n'
196              elif '<p><b><u>Tests/1 Million</u></b><br' in line:
197                  newstr += '                    <p><b><u>Tests/1 Million</u></b><br><br>' + \
198                      country_numbers[8] + '</p>\n'
199              elif '<p><b><u>Population</u></b><br><br>' in line:
200                  newstr += '                    <p><b><u>Population</u></b><br><br>' + \
201                      country_numbers[9] + '</p>\n'
202              else:
203                  newstr += line
204      f.close()
205  
206      # update html file by writing data to file
207      with open(dir_path, 'w') as f:
208          f.write(newstr)
209      f.close()
210  
211      return country_numbers

214  def update_index_html(country: str):
215      ''' 
216      updates the graph on main index page as well as the data in the table of the top 25 most affect countries 
217      :param country: country to be updated in the table on the home index page 
218      :return: void 
219      '''
220      titled_country = country.title()
221      req = Request('https://www.worldometers.info/coronavirus/',
222                    headers={'User-Agent': 'Mozilla/5.0'})
223      page = urlopen(req).read()
224      page = BeautifulSoup(page, 'html.parser')
225  
226      # get a list of the main numbers using html 'class="maincounter-number"'
227      # separator
228      world_numbers_html = page.findAll("div", class_="maincounter-number")
229      world_numbers = []
230  
231      for number in world_numbers_html:
232          # use regex to extract numbers from world_numbers_html, put those
233          # numbers in world_numbers list
234          finder = re.compile(r'\b\d[\d,.]*\b')
235          for match in finder.finditer(str(number)):
236              world_numbers.append(match.group())
237  
238      date_aot = str(datetime.today())[0:10]
239      dir_path = os.path.dirname(
240          os.path.realpath(__file__))[
241          0:56] + "/index.html"
242      with open(dir_path, 'r') as f:
243          newstr = """"""
244          for line in f.readlines():
245              # update main info for world
246              if "<h3>World Cases:" in line:
247                  newstr += '                   <h3>World Cases: <span style="color: blue;">' + world_numbers[0] + \
248                            '</span> &emsp; &emsp; &emsp; Deaths: <span style="color: red;">' + \
249                      world_numbers[1] + \
250                            '</span> &emsp; &emsp; &emsp; Recovered: <span style="color: rgb(15, 146, 15);">' + \
251                            world_numbers[2] + '</span></h3>\n'
252              # update date for pulling graph .png file
253              elif '<img src="src/graphs/world/world_allfour_' in line:
254                  newstr += '                    <img src="src/graphs/world/world_allfour_' + \
255                      date_aot + '.png" width="1100"/> <br><br><br>\n'
256              else:
257                  newstr += line
258      f.close()
259      # update html page by writing data to file
260      with open(dir_path, 'w') as f:
261          f.write(newstr)
262      f.close()
263  
264      # update data in table on main index.html page
265      country_numbers = get_country_data(country)
266      with open(dir_path, 'r') as f:
267          newstr = """"""
268          inlines = False
269          count = -1
270          for line in f.readlines():
271              if ('<td class="tg-0pky"><b>' + country) in str(line):
272                  inlines = True
273              if count == 10:
274                  inlines = False
275              # when in desired lines, update value depending on given category
276              if inlines:
277                  if count == -1:
278                      newstr += '                           <td class="tg-0pky"><b>' + titled_country + '</b></td>\n'
279                  elif count == 0:
280                      newstr += '                           <td class="tg-0lax" id="totalcases">' + \
281                          country_numbers[count] + '</td>' + '\n'
282                  elif count == 1:
283                      newstr += '                           <td class="tg-0lax" id="newcases">+' + \
284                          country_numbers[count] + '</td>' + '\n'
285                  elif count == 2:
286                      newstr += '                           <td class="tg-0lax" id="casespermil">' + \
287                          country_numbers[5] + '</td>' + '\n'
288                  elif count == 3:
289                      newstr += '                           <td class="tg-0lax" id="totaldeaths">' + \
290                          country_numbers[2] + '</td>' + '\n'
291                  elif count == 4:
292                      newstr += '                           <td class="tg-0lax" id="newdeaths">+' + \
293                          country_numbers[3] + '</td>' + '\n'
294                  elif count == 5:
295                      newstr += '                           <td class="tg-0lax" id="deathspermil">' + \
296                          country_numbers[6] + '</td>' + '\n'
297                  elif count == 6:
298                      newstr += '                           <td class="tg-0lax" id="totalrecovered">' + \
299                          country_numbers[4] + '</td>' + '\n'
300                  elif count == 7:
301                      newstr += '                           <td class="tg-0lax" id="totaltests">' + \
302                          country_numbers[count] + '</td>' + '\n'
303                  elif count == 8:
304                      newstr += '                           <td class="tg-0lax" id="testspermil">' + \
305                          country_numbers[count] + '</td>' + '\n'
306                  elif count == 9:
307                      newstr += '                           <td class="tg-0lax" id="population">' + \
308                          country_numbers[count] + '</td>' + '\n'
309                  else:
310                      newstr += '                           <td class="tg-0lax">' + \
311                          country_numbers[count] + '</td>' + '\n'
312                  count += 1
313              else:
314                  newstr += line
315      f.close()
316      # update html page page by writing to file
317      with open(dir_path, 'w') as f:
318          f.write(newstr)
319      f.close()
320  
321      print(country + str(country_numbers))
